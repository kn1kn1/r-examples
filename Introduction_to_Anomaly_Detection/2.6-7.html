# MASSroad=
library(MASS)

X <- road / road$drivers # drivers()
X <- as.matrix(log(X[, -2] + 1)) # 

mx <- colMeans(X) # 

# `%*%` 
# `matrix(1, nrow(X), 1) %*% mx`
# X
Xc <- X - matrix(1, nrow(X), 1) %*% mx # 

# 
# t()`t(Xc) %*% Xc` (x' - u)
Sx <- t(Xc) %*% Xc / nrow(X) # 

# 1
# p.43 `ncol(X)` 1
# a <- rowSums((Xc %*% solve(Sx)) * Xc) / ncol(X) # 1
# 
a <- colSums(t(Xc) * solve(Sx, t(Xc))) / ncol(X) # 1

plot(a,
     xlab = "index",
     ylab = "anomaly score",
     ylim = c(-1, 30) / ncol(X))
lines(0:30, rep(1, length(0:30)), col = "red", lty = 2) # 1
