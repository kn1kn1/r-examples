library(kernlab)
library(MASS)
cc <-
  c(
    "Min.Price",
    "Price",
    "Max.Price",
    "MPG.city",
    "MPG.highway",
    "EngineSize",
    "Horsepower",
    "RPM",
    "Rev.per.mile",
    "Fuel.tank.capacity",
    "Length",
    "Wheelbase",
    "Width",
    "Turn.circle",
    "Weight"
  )
mask <- is.element(colnames(Cars93), cc) # 15
Xc <- t(scale(Cars93[, mask])) # 
colnames(Xc) <- t(Cars93[, "Make"]) # (make)

m <- 2 # 
sig <- 0.1 # 
li <- c(-6, 7) # 

# kpca 5.4(p.151)1
kpc <-
  kpca(
    t(Xc),
    kernel = "rbfdot",
    kpar = list(sigma = sig),
    features = m
  )
Zt <- rotated(kpc) # 

# 
plot(
  Zt[, 1],
  Zt[, 2],
  xlab = "1st PC",
  ylab = "2nd PC",
  cex = 3,
  col = 3,
  xlim = li,
  ylim = li,
  main = paste0("sig: ", sig)
)

# 
text(Zt[, 1],
     Zt[, 2],
     c(1:93),
     cex = 0.8,
     xlim = li,
     ylim = li)
